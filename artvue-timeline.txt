let lang = 'zh';
let civs = { jp: true, wl: true };
let yr = -2100;
let expId = null;

function setLang(l) {
  lang = l;
  document.querySelectorAll('.lb').forEach(b => b.classList.remove('on'));
  document.querySelector(`.lb[onclick="setLang('${l}')"]`).classList.add('on');
  
  document.getElementById('htitle').textContent = UI[l].htitle;
  document.getElementById('hsub').textContent = UI[l].hsub;
  
  const btnJp = document.querySelector('.cb[data-c="jp"]');
  const btnWl = document.querySelector('.cb[data-c="wl"]');
  btnJp.textContent = (l==='en') ? '🇯🇵 Japan' : '🇯🇵 日本';
  btnWl.textContent = (l==='en') ? '🌍 World' : '🌍 世界';

  updateScrubberText();
  render();
}

function toggleCiv(c) {
  civs[c] = !civs[c];
  document.querySelector(`.cb[data-c="${c}"]`).classList.toggle('on', civs[c]);
  render();
}

function updateScrubberText() {
  const y = yr;
  const t = y < 0 ? `${Math.abs(y)} BC` : `${y} AD`;
  document.getElementById('syr').textContent = `${t} ${UI[lang].drag}`;
}

function render() {
  const container = document.getElementById('mc');
  let h = '';
  
  const items = DATA.filter(d => d.e >= yr);
  
  if(items.length === 0) {
    container.innerHTML = `<div style="text-align:center;color:var(--ink60);padding:40px;">${UI[lang].empty}</div>`;
    return;
  }

  items.forEach(d => {
    const isExp = expId === d.id;
    const t = d[lang];
    
    h += `<div class="dc ${isExp ? 'exp' : ''}" onclick="toggleExp(event, '${d.id}')">
            <div class="dc-head">
              <div>
                <div class="dc-name">${t.name}</div>
                <div class="dc-yrs">${t.yrs}</div>
              </div>
              <div class="dc-tag">${t.tag}</div>
            </div>
            
            <div class="detail" onclick="event.stopPropagation()">
              <div class="d-rule"></div>
              <div class="d-desc">${t.desc}</div>
              
              <div class="info-grid">
                <div class="info-box">
                  <h4>${UI[lang].emp}</h4>
                  <p>${t.emps}</p>
                </div>
                <div class="info-box">
                  <h4>${UI[lang].event}</h4>
                  <p>${t.events}</p>
                </div>
              </div>
              
              <div class="art-sec-title">${UI[lang].art}</div>
              <div class="art-scroll">`;
              
    d.art.forEach((a, idx) => {
      const at = a[lang];
      h += `  <div class="art-card" onclick="showArt('${d.id}', ${idx})">
                <div class="art-img-wrap"><img src="${a.img}" alt="${at.name}" loading="lazy"></div>
                <div class="art-card-info">
                  <div class="art-card-title">${at.name}</div>
                  <div class="art-card-sub">${at.where}</div>
                </div>
              </div>`;
    });
              
    h += `    </div>
              <div class="d-rule" style="margin-top:20px;"></div>`;
              
    if (civs.jp || civs.wl) {
      h += `  <div class="parallel-wrap">`;
      if (civs.jp) {
        h += `<div class="p-box p-jp">
                <div class="p-head">🇯🇵 ${lang==='en'?'Japan':lang==='ja'?'日本':'同期日本'}</div>
                <div class="p-text">${t.jp}</div>
              </div>`;
      }
      if (civs.wl) {
        h += `<div class="p-box p-wl">
                <div class="p-head">🌍 ${lang==='en'?'World':lang==='ja'?'世界':'同期世界'}</div>
                <div class="p-text">${t.wl}</div>
              </div>`;
      }
      h += `  </div>`;
    }
              
    h += `  </div>
          </div>`;
  });
  
  container.innerHTML = h;
}

function toggleExp(e, id) {
  if (expId === id && e.target.closest('.detail')) return;
  expId = expId === id ? null : id;
  render();
  if(expId) {
    setTimeout(() => {
      const el = document.querySelector('.dc.exp');
      if(el) {
        const y = el.getBoundingClientRect().top + window.pageYOffset - 90;
        window.scrollTo({top: y, behavior: 'smooth'});
      }
    }, 150);
  }
}

function showArt(did, ai) {
  const d = DATA.find(x => x.id === did);
  const a = d.art[ai];
  const at = a[lang];
  
  document.getElementById('mimg').src = a.img;
  document.getElementById('mtitle').textContent = at.name;
  document.getElementById('mwhere').textContent = at.where;
  document.getElementById('mdesc').textContent = at.desc;
  
  document.getElementById('modal').classList.add('show');
}

function closeM() {
  document.getElementById('modal').classList.remove('show');
  setTimeout(() => { document.getElementById('mimg').src = ''; }, 200);
}

document.getElementById('modal').addEventListener('click', e => {
  if (e.target === document.getElementById('modal')) closeM();
});

const sl = document.getElementById('sl');
sl.addEventListener('input', () => {
  yr = +sl.value;
  updateScrubberText();
  expId = null; 
  render();
});

setLang('zh');
